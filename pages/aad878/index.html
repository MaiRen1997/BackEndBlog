中间件的使用add_middleware

# 什么是中间件

中间件是在请求到达路由处理函数之前和响应返回客户端之前执行的函数，可以在处理流程中插入自定义逻辑。

# 基本作用

  1. 请求预处理：日志记录、身份验证、请求解析
  2. 响应后处理：添加响应头、压缩响应、错误处理
  3. 拦截请求：权限检查、限流、请求过滤
  4. 性能监控：请求耗时统计
  5. 中间件无法获取到路径参数：如/{user_id}中的 user_id

# 基本语法

# 使用add_middleware添加类中间件

  1. 使用add_middleware添加中间件
  2. 使用CORSMiddleware跨域中间件
from fastapi import FastAPI
from starlette.middleware.base import BaseHTTPMiddleware
from fastapi.middleware.cors import CORSMiddleware
app = FastAPI()

# 方式1：使用内置中间件
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 方式2：自定义中间件
class CustomMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        # 请求前处理
        print("Before request")
        
        # 调用下一个中间件或路由处理函数
        response = await call_next(request)
        
        # 响应后处理
        print("After request")
        
        return response

app.add_middleware(CustomMiddleware)

# 类中间件传参

  1. 通过__init__ 传递参数
  2. 必须调用父类初始化，传入 app
  3. app 参数位置固定
class AuthMiddleware(BaseHTTPMiddleware):
    def __init__(self, app, secret_key: str, excluded_paths: list = None):
        super().__init__(app)  # ⚠️ 必须调用父类初始化，传入 app
        self.secret_key = secret_key
        self.excluded_paths = excluded_paths or []
    
    async def dispatch(self, request: Request, call_next):
        # 使用传入的参数
        if request.url.path in self.excluded_paths:
            return await call_next(request)
        
        token = request.headers.get("Authorization")
        if token != self.secret_key:
            return JSONResponse(
                status_code=401,
                content={"error": "Invalid token"}
            )
        
        return await call_next(request)

# 注册时传参
app.add_middleware(
    AuthMiddleware, 
    secret_key="my-secret-123", 
    excluded_paths=["/login", "/health", "/docs"]
)

# websocket中间件

WebSocket 中间件必须用类实现

from fastapi import FastAPI, WebSocket
from starlette.types import ASGIApp, Receive, Scope, Send

app = FastAPI()

# WebSocket 中间件必须用类实现
class WebSocketLoggingMiddleware:
    def __init__(self, app: ASGIApp):
        self.app = app
    
    async def __call__(self, scope: Scope, receive: Receive, send: Send):
        if scope["type"] == "websocket":
            print(f"WebSocket 连接: {scope.get('path')}")
            # 可以修改或包装 send/receive
            await self.app(scope, receive, send)
        else:
            await self.app(scope, receive, send)

app.add_middleware(WebSocketLoggingMiddleware)

@app.websocket("/ws/{client_id}")
async def websocket_endpoint(websocket: WebSocket, client_id: str):
    await websocket.accept()
    await websocket.send_text(f"Hello {client_id}")

# 执行顺序

中间件的执行顺序与添加顺序相反

from fastapi import FastAPI
from starlette.middleware.base import BaseHTTPMiddleware
import time

class MiddlewareA(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        print("A: Before")
        response = await call_next(request)
        print("A: After")
        return response

class MiddlewareB(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        print("B: Before")
        response = await call_next(request)
        print("B: After")
        return response

app = FastAPI()

# 添加顺序：先A，后B
app.add_middleware(MiddlewareA)
app.add_middleware(MiddlewareB)

@app.get("/")
async def root():
    print("Route handler")
    return {"message": "Hello"}

# 执行顺序：
# 请求 → A:Before → B:Before → Route handler → B:After → A:After → 响应
# 输出：
# A: Before
# B: Before
# Route handler
# B: After
# A: After

# 参数分解

# request

request 实际上是 starlette.requests.Request 对象，FastAPI 只是重新导出了它。

# 主要属性和方法

from fastapi import FastAPI, Request
from starlette.middleware.base import BaseHTTPMiddleware

class RequestInfoMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        # 1. 请求基本信息
        print(f"Method: {request.method}")           # GET, POST等
        print(f"URL: {request.url}")                 # 完整URL
        print(f"Path: {request.url.path}")           # 路径部分
        print(f"Query params: {request.url.query}")  # 查询参数
        
        # 2. 请求头
        headers = request.headers
        user_agent = headers.get("user-agent")
        auth = headers.get("authorization")
        print(f"Headers: {dict(headers)}")
        
        # 3. 客户端信息
        client_host = request.client.host
        client_port = request.client.port
        print(f"Client: {client_host}:{client_port}")
        
        # 4. 请求体（注意：只能读取一次！）
        # 方式1：直接读取
        body = await request.body()
        print(f"Body: {body}")
        
        # 方式2：作为JSON解析
        try:
            json_body = await request.json()
            print(f"JSON: {json_body}")
        except:
            pass
        
        # 5. 路径参数和查询参数
        # 注意：这些通常需要在路由处理函数中获取
        # 在中间件中无法直接获取已解析的参数
        
        # 6. 自定义状态存储
        request.state.user_id = "123"  # 存储数据供后续使用
        request.state.request_time = datetime.now()
        
        response = await call_next(request)
        return response

# 在路由中可以访问
@app.get("/test")
async def test_endpoint(request: Request):
    user_id = request.state.user_id  # 获取中间件存储的数据
    return {"user_id": user_id}

# call_next

具体来说，它是一个异步函数：输入：Request 对象，输出：Response 对象的协程

# 作用

class CallNextDemoMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        # 1. 调用下一个中间件或路由处理函数
        # 这是整个中间件最关键的调用
