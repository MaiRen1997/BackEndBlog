跨域问题解决

# FastAPI 解决跨域问题详解

FastAPI使用CORS中间件来解决跨域问题。下面是完整的解决方案和最佳实践。

# 一、什么是跨域问题

浏览器的同源策略限制：协议、域名、端口不同时，JavaScript无法访问另一个域的资源。

跨域场景示例：

  • 前端：http://localhost:3000
  • 后端：http://localhost:8000
  • 端口不同 → 触发跨域

# 二、基础解决方案（CORS中间件）

# 1. 最简单的配置（允许所有）

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI()

# 配置CORS中间件
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],        # 允许所有源（不推荐生产环境）
    allow_credentials=True,      # 允许携带凭证（cookie等）
    allow_methods=["*"],         # 允许所有HTTP方法
    allow_headers=["*"],         # 允许所有请求头
)

@app.get("/")
async def main():
    return {"message": "Hello World"}

# 2. 生产环境推荐配置（指定源）

from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI()

# 定义允许的源列表
origins = [
    "http://localhost:3000",      # React开发服务器
    "http://localhost:8080",      # Vue开发服务器
    "https://yourdomain.com",     # 生产域名
    "https://api.yourdomain.com", # API域名
    "http://127.0.0.1:3000",     # 本地IP
]

app.add_middleware(
    CORSMiddleware,
    allow_origins=origins,         # 只允许特定源
    allow_credentials=True,        # 允许携带凭证
    allow_methods=["GET", "POST", "PUT", "DELETE", "OPTIONS"],  # 允许的方法
    allow_headers=["Content-Type", "Authorization", "X-Requested-With"],  # 允许的请求头
    expose_headers=["X-Custom-Header"],  # 暴露给前端的响应头
    max_age=3600,                  # 预检请求缓存时间（秒）
)

@app.get("/")
async def main():
    return {"message": "Hello World"}

# 三、完整示例（带详细配置）

from fastapi import FastAPI, Header, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import uvicorn

app = FastAPI(title="CORS示例API")

# ==================== CORS配置 ====================
# 配置选项详解
ALLOWED_ORIGINS = [
    "http://localhost:3000",
    "http://127.0.0.1:3000",
    "http://localhost:8080",
    "https://example.com",
]

# 根据环境动态配置
import os
ENVIRONMENT = os.getenv("ENVIRONMENT", "development")

if ENVIRONMENT == "production":
    # 生产环境：严格限制
    origins = ALLOWED_ORIGINS
    allow_all_methods = False
else:
    # 开发环境：可以宽松一些
    origins = ["*"] if ENVIRONMENT == "development" else ALLOWED_ORIGINS
    allow_all_methods = True

app.add_middleware(
    CORSMiddleware,
    allow_origins=origins,
    allow_credentials=True,
    allow_methods=["GET", "POST", "PUT", "DELETE", "OPTIONS", "PATCH"] if not allow_all_methods else ["*"],
    allow_headers=[
        "Content-Type",
        "Authorization",
        "Accept",
        "X-Requested-With",
        "X-Custom-Header",
    ],
    expose_headers=["X-Total-Count", "X-Page"],
    max_age=600,  # 10分钟
)

# ==================== 业务接口 ====================
class User(BaseModel):
    name: str
    email: str

@app.get("/")
async def root():
    """公开接口"""
    return {
        "message": "CORS已配置",
        "environment": ENVIRONMENT,
        "allowed_origins": origins if origins != ["*"] else "所有源（开发模式）"
    }

@app.get("/users")
async def get_users():
    """获取用户列表"""
    return {
        "users": [
            {"id": 1, "name": "张三"},
            {"id": 2, "name": "李四"},
        ],
        "total": 2
    }

@app.post("/users")
async def create_user(user: User):
    """创建用户"""
    return {
        "message": "用户创建成功",
        "user": user
    }

@app.get("/protected")
async def protected_route(authorization: str = Header(None)):
    """需要认证的接口"""
    if not authorization:
        raise HTTPException(status_code=401, detail="未授权")
    
    return {
        "message": "认证成功",
        "token": authorization[:20] + "..." if len(authorization) > 20 else authorization
    }

# 自定义OPTIONS处理（通常不需要，CORSMiddleware已处理）
@app.options("/custom-options")
async def options_route():
    """自定义OPTIONS方法"""
    return {"message": "OPTIONS请求成功"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000, reload=True)
