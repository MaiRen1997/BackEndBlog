依赖注入与安全认证

# 依赖注入

# 依赖注入系统

from fastapi import Depends, HTTPException

# 定义依赖（如Token验证）
def verify_token(token: str = Query(...)):
    if token != "secret123":
        raise HTTPException(status_code=401, detail="Invalid token")
    return True

# 在路由中使用依赖
@app.get("/secure-data", dependencies=[Depends(verify_token)])
async def get_secure_data():
    return {"data": "敏感数据"}

# 在函数参数中使用依赖
@app.get("/user-profile")
async def get_profile(auth: bool = Depends(verify_token)):
    return {"profile": "用户资料"}

# 安全认证与JWT

# 说明

  • OAuth2密码流
  • JWT Token生成与验证
  • 密码哈希（使用passlib）
from fastapi.security import OAuth2PasswordBearer
from jose import JWTError, jwt

SECRET_KEY = "your-secret-key"
ALGORITHM = "HS256"

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

def get_current_user(token: str = Depends(oauth2_scheme)):
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        return payload.get("sub")
    except JWTError:
        raise HTTPException(status_code=401)

# 数据库集成

异步ORM推荐：Tortoise-ORM 或 SQLAlchemy 1.4+ (asyncio)

# Tortoise-ORM示例
from tortoise import fields, models

class User(models.Model):
    id = fields.IntField(pk=True)
    name = fields.CharField(max_length=50)
    email = fields.CharField(max_length=100, unique=True)
    
    class Meta:
        table = "users"

# 在路由中使用
@app.get("/users/{user_id}")
async def get_user(user_id: int):
    user = await User.get(id=user_id)
    return {"name": user.name, "email": user.email}

# 中间件与CORS

from fastapi.middleware.cors import CORSMiddleware

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # 生产环境应指定具体域名
    allow_methods=["*"],
    allow_headers=["*"],
)

# 自定义中间件
@app.middleware("http")
async def log_requests(request, call_next):
    print(f"请求路径: {request.url.path}")
    response = await call_next(request)
    return response

# 异常处理与统一响应

from fastapi import HTTPException
from fastapi.responses import JSONResponse

# 统一响应结构
def api_response(code=0, message="success", data=None):
    return {"code": code, "message": message, "data": data}

# 全局异常处理
@app.exception_handler(HTTPException)
async def http_exception_handler(request, exc):
    return JSONResponse(
        status_code=exc.status_code,
        content=api_response(code=exc.status_code, message=exc.detail)
    )
