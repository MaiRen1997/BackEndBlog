# 子路由

# routers/users.py
from fastapi import APIRouter

router = APIRouter()

@router.get("/users/")
async def get_users():
    return {"users": ["Alice", "Bob"]}

@router.get("/users/{user_id}")
async def get_user(user_id: int):
    return {"user_id": user_id, "name": f"User {user_id}"}

# 主路由

使用include_router关键字，合并子路由

# main.py
from fastapi import FastAPI
from routers import users

app = FastAPI()

# 包含用户路由
app.include_router(
    users.router,                     # 必需：APIRouter实例
    prefix="/api",                   # 可选：路由前缀
    tags=["users", "admin"],         # 可选：OpenAPI标签
    dependencies=[Depends(auth)],    # 可选：全局依赖
    responses={404: {"description": "Not found"}},  # 可选：全局响应
    include_in_schema=True           # 可选：是否包含在OpenAPI中
)

# 路由优先级

FastAPI 按照路由注册的顺序匹配请求：先注册的优先级更高 ,先注册的会覆盖掉后注册的

也就是路由匹配，先匹配第一个，匹配不到，才会继续往后匹配

因此，在路由匹配时，需要先注册详细的路由，再注册通用路由

# 示例

  1. /items/special → 匹配第一个路由 (返回 "Item ID as string: special")
  2. 永远不会进入第二个路由！因为 "special" 被当作 item_id 参数处理了
  3. 为了解决这个情况，需要将两个路由的顺序进行调换
from fastapi import FastAPI

app = FastAPI()

# 场景A：先注册通用路由
@app.get("/items/{item_id}")
async def get_item(item_id: str):
    return {"message": f"Item ID as string: {item_id}"}

# 后注册特定路由
@app.get("/items/special")
async def get_special_item():
    return {"message": "Special item"}

# 测试结果：
# GET /items/special → 匹配第一个路由 ✅ (返回 "Item ID as string: special")
# 永远不会进入第二个路由！因为 "special" 被当作 item_id 参数处理了

# 不同类型参数的匹配

  1. 如下代码：
    1. 如果是me，则匹配第一个路由
    2. 如果类型是数字，则匹配第二个路由
    3. 如果类型不是me，且类型是字符串，则匹配第三个数字
from fastapi import FastAPI

app = FastAPI()

# 注册顺序1：静态路由
@app.get("/users/me")
async def get_current_user():
    return {"message": "This is the current user"}

# 注册顺序2：动态路由
@app.get("/users/{user_id}")
async def get_user(user_id: int):
    return {"message": f"User {user_id}"}

# 注册顺序3：更通用的路由
@app.get("/users/{any_path}")
async def get_any_user(any_path: str):
    return {"message": f"Any path: {any_path}"}

# 测试结果：
# GET /users/me     → 匹配第一个路由 ✅ (返回 current user)
# GET /users/123    → 匹配第二个路由 ✅ (返回 User 123)
# GET /users/abc    → 匹配第二个路由失败(需要int)，然后匹配第三个 ✅ (返回 Any path: abc)
