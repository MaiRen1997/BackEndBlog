Python中Request的使用

# Request使用

import requests

BASE_URL = "http://127.0.0.1:8000"

# 路径参数
response = requests.get(f"{BASE_URL}/items/456?q=hello")
print("路径参数:", response.json())

# 查询参数
response = requests.get(
    f"{BASE_URL}/search",
    params={"q": "fastapi", "page": 3, "size": 15}
)
print("查询参数:", response.json())

# 请求体
response = requests.post(
    f"{BASE_URL}/users",
    json={
        "username": "李四",
        "age": 30,
        "email": "lisi@example.com",
        "tags": ["fastapi", "python"]
    }
)
print("请求体:", response.json())

# 表单数据
response = requests.post(
    f"{BASE_URL}/login",
    data={
        "username": "王五",
        "password": "secret123",
        "remember_me": "true"
    }
)
print("表单数据:", response.json())

# 请求头
response = requests.get(
    f"{BASE_URL}/protected",
    headers={
        "Authorization": "Bearer my-token-xyz",
        "X-Request-ID": "req-001",
        "Accept-Language": "en-US"
    }
)
print("请求头:", response.json())
