# 生成随机密码

import random
import string

def generate_password(length=8):
    characters = string.ascii_letters + string.digits + string.punctuation
    password = ''.join(random.choice(characters) for _ in range(length))
    return password

print(generate_password(10))  # 例如：'aB3$kL9@mN'

    results = [random.randint(1, 6) for _ in range(times)]
    return results

# 7. 常用方法总结

方法 说明 示例
random() 0.0到1.0的随机浮点数 random.random()
randint(a, b) a到b的随机整数 random.randint(1, 10)
uniform(a, b) a到b的随机浮点数 random.uniform(1.5, 5.5)
choice(seq) 从序列中随机选一个 random.choice([1,2,3])
sample(seq, k) 从序列中随机选k个不重复的 random.sample(range(100), 5)
shuffle(seq) 打乱序列顺序 random.shuffle(my_list)
seed(x) 设置随机种子 random.seed(42)
