## 路由分模块

### 主、子路由

#### 子路由

```python
# routers/users.py
from fastapi import APIRouter

router = APIRouter()

@router.get("/users/")
async def get_users():
    return {"users": ["Alice", "Bob"]}

@router.get("/users/{user_id}")
async def get_user(user_id: int):
    return {"user_id": user_id, "name": f"User {user_id}"}
```

#### 主路由

```python
# main.py
from fastapi import FastAPI
from routers import users

app = FastAPI()

# 包含用户路由
app.include_router(
    users.router,                     # 必需：APIRouter实例
    prefix="/api",                   # 可选：路由前缀
    tags=["users", "admin"],         # 可选：OpenAPI标签
    dependencies=[Depends(auth)],    # 可选：全局依赖
    responses={404: {"description": "Not found"}},  # 可选：全局响应
    include_in_schema=True           # 可选：是否包含在OpenAPI中
)

```

### 嵌套路由

#### 最里层

```python
# routers/admin.py
from fastapi import APIRouter

router = APIRouter()

@router.get("/dashboard/")
async def admin_dashboard():
    return {"message": "Admin Dashboard"}

```

#### 中间层

```python
# routers/users.py
from fastapi import APIRouter
from .admin import router as admin_router

router = APIRouter()

# 嵌套包含其他路由
router.include_router(
    admin_router,
    prefix="/admin",
    tags=["admin"]
)

@router.get("/profile/")
async def user_profile():
    return {"message": "User Profile"}

```

#### 最外层

```python
# main.py
from fastapi import FastAPI
from routers import users

app = FastAPI()

# 最终路径: /users/profile/, /users/admin/dashboard/
app.include_router(users.router, prefix="/users", tags=["users"])

```

## 中间件&跨域

### 基础使用

```python
from fastapi import FastAPI
from starlette.middleware.base import BaseHTTPMiddleware
from fastapi.middleware.cors import CORSMiddleware
app = FastAPI()

# 方式1：使用内置中间件
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 方式2：自定义中间件
class CustomMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        # 请求前处理
        print("Before request")
        
        # 调用下一个中间件或路由处理函数
        response = await call_next(request)
        
        # 响应后处理
        print("After request")
        
        return response

app.add_middleware(CustomMiddleware)
```

### 执行顺序

```python
from fastapi import FastAPI
from starlette.middleware.base import BaseHTTPMiddleware
import time

class MiddlewareA(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        print("A: Before")
        response = await call_next(request)
        print("A: After")
        return response

class MiddlewareB(BaseHTTPMiddleware):
    async def dispatch(self, request, call_next):
        print("B: Before")
        response = await call_next(request)
        print("B: After")
        return response

app = FastAPI()

# 添加顺序：先A，后B
app.add_middleware(MiddlewareA)
app.add_middleware(MiddlewareB)

@app.get("/")
async def root():
    print("Route handler")
    return {"message": "Hello"}

# 执行顺序：
# 请求 → A:Before → B:Before → Route handler → B:After → A:After → 响应
# 输出：
# A: Before
# B: Before
# Route handler
# B: After
# A: After
```

## Depends

### 基础使用

```python
from fastapi import Query

def query_dependency(
    skip: int = Query(0, alias="offset"),  # 客户端发送 ?offset=10
    limit: int = Query(100, alias="pageSize")  # 客户端发送 ?pageSize=20
):
    return {"skip": skip, "limit": limit}

# 客户端请求: /items?offset=10&pageSize=20
@app.get("/items")
def get_items(params: dict = Depends(query_dependency)):
    return params  # 返回 {"skip": 10, "limit": 20}
```
